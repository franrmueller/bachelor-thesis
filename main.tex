\section*{Citation Style Examples (APA \& biblatex)}

\begin{itemize}
  \item \textbf{Standard citation (parenthetical)} \\
  Use when the source supports your point but isn't central to the sentence.\\
  \textit{Example:} AI is foundational in modern computing \parencite{russell_artificial_2010}.

  \item \textbf{With page number} \\
  Use when referring to a specific passage or detail.\\
  \textit{Example:} AI systems raise ethical questions \parencite[p.~45]{russell_artificial_2010}.

  \item \textbf{With page range} \\
  Use when summarizing or referencing a larger section.\\
  \textit{Example:} Search algorithms are discussed in detail \parencite[pp.~15--24]{russell_artificial_2010}.

  \item \textbf{Indirect citation with ``cf.''} \\
  Use when comparing or referencing related work (not quoted directly).\\
  \textit{Example:} Other researchers take a similar view \parencite[cf.][p.~112]{feuerriegel_generative_2024}.

  \item \textbf{Narrative citation} \\
  Use when the author is part of your sentence.\\
  \textit{Example:} \textcite{feuerriegel_generative_2024} discuss the impact of generative AI.

  \item \textbf{Narrative citation with page number} \\
  Combine author-as-subject with specific location in the text.\\
  \textit{Example:} \textcite[p.~112]{feuerriegel_generative_2024} emphasize model transparency.

  \item \textbf{Narrative citation with ``cf.''} \\
  Used in comparative framing, with the author mentioned.\\
  \textit{Example:} \textcite[cf.][p.~112]{feuerriegel_generative_2024} provide a related perspective.

  \item \textbf{Multiple citations with page numbers} \\
  Use to support a general claim with multiple sources.\\
  \textit{Example:} AI in finance is widely explored 
  \parencites
    [pp.~15--17]{russell_artificial_2010}
    [p.~112]{feuerriegel_generative_2024}.

  \item \textbf{Quote with citation} \\
  Use for exact wording with mandatory page reference.\\
  \textit{Example:} ``Artificial intelligence is the new electricity'' \parencite[p.~7]{russell_artificial_2010}.
\end{itemize}
\end{titlepage}
